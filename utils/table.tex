% table
\newcommand*{\customtable}[5]{
    \begin{table}[b!]
        \caption{#2}
        \begin{adjustwidth}{-1.5cm}{-1.5cm}  
        \begin{center}
            
           #4{#1}
                \hline

                #3
            #5

        \end{center}
        \end{adjustwidth}
    \end{table}
}

\newcommand*{\standardtable}[3]{
    \customtable{#1}{#2}{#3}{\begin{tabular}}{\end{tabular}}
}

\newcommand*{\wraptable}[3]{
    \customtable{#1}{#2}{#3}{\begin{tabularx}{\textwidth}}{\end{tabularx}}
}

% rows
\newcommand*{\tabletitle}[1]{\textbf{#1}}
\newcommand*{\twocols}[2]{

\newcommand*{\threecols}[3]{

    \hline
}
\newcommand*{\threecolsnl}[3]{

\newcommand*{\threecolsdel}[3]{\twocols{#1}{#2}}
\newcommand*{\fivecols}[5]{
    #1 & #2 & #3 & #4 & #5 \\
    \hline
}
