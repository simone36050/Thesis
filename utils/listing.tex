\newcommand{\pagelisting}[5]{
    % \newgeometry{top=-0.7cm,bottom=-7.5cm,left=-7.5cm,right=10.5cm}

    % \SetSinglespace{-0.5}

    \begin{figure}[#3]
    \caption{#1}
    \renewcommand{\labelitemi}{}
    #2
    \setlist{nosep,leftmargin=*}
    \setlist[1]{itemsep=0.8em}

    \begin{framed}
        \begin{minipage}[t]{.48\textwidth}
            #4
        \end{minipage}
        \begin{minipage}[t]{.48\textwidth}
            #5
        \end{minipage}
    \end{framed}
    \end{figure}

    \restoregeometry
}
